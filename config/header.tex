% Page geometry with standard margins
\usepackage[a4paper,top=2.5cm,bottom=2.5cm,left=2.5cm,right=2.5cm]{geometry}

% Font and typography with Unicode support (Tectonic uses XeTeX)
\usepackage{fontspec}
\setmainfont{Arial}           % Widely available sans-serif font
\setsansfont{Arial}           % Consistent sans-serif for headings

% Simple symbols instead of icons (more compatible)
\usepackage{tikz}
\usepackage{amssymb}  % Additional math symbols

% Enhanced Unicode support for Greek letters and special characters
\usepackage{amsmath,amssymb}  % Math symbols including Greek letters
\usepackage{gensymb}  % Additional symbols
\usepackage{mathrsfs}  % Math script fonts

\usepackage{microtype}  % Improves typography

% Make bold text even more prominent
\renewcommand{\bfdefault}{b}  % Use bold weight
\renewcommand{\mddefault}{m}  % Use medium weight for regular text

% Colors
\usepackage{xcolor}
\definecolor{headingcolor}{RGB}{0,102,204}      % Rich royal blue
\definecolor{subheadingcolor}{RGB}{30,144,255}  % Vibrant dodger blue
\definecolor{emphasiscolor}{RGB}{220,50,47}     % Rich red
\definecolor{codebackground}{RGB}{245,245,245}  % Light gray
\definecolor{linkblue}{RGB}{0,123,255}          % Bright link blue

% Headers and sections with symbols
\usepackage{titlesec}
\titleformat{\section}
  {\Large\bfseries\color{headingcolor}}{\thesection}{1em}{}
\titleformat{\subsection}
  {\large\bfseries\color{subheadingcolor}}{\thesubsection}{1em}{}
\titleformat{\subsubsection}
  {\normalsize\bfseries\color{subheadingcolor}}{\thesubsubsection}{1em}{}

% Spacing
\titlespacing*{\section}{0pt}{2ex plus 1ex minus 0.2ex}{1ex plus 0.2ex}
\titlespacing*{\subsection}{0pt}{1.5ex plus 1ex minus 0.2ex}{0.8ex plus 0.2ex}
\titlespacing*{\subsubsection}{0pt}{1.2ex plus 0.8ex minus 0.2ex}{0.5ex plus 0.2ex}

% Paragraph formatting
\usepackage{parskip}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.8em}

% Lists with symbols
\usepackage{enumitem}
\setlist{itemsep=0.3em,topsep=0.5em,parsep=0em}
\renewcommand{\labelitemi}{\textbullet}  % Bullet points
\renewcommand{\labelitemii}{\textopenbullet}  % Nested bullets
\renewcommand{\labelitemiii}{\textbullet}  % Third level bullets

% Code blocks and verbatim
\usepackage{fancyvrb}
\usepackage{listings}
\lstset{
  backgroundcolor=\color{codebackground},
  basicstyle=\small\ttfamily,
  breaklines=true,
  frame=single,
  xleftmargin=0.5cm,
  xrightmargin=0.5cm,
  framexleftmargin=3mm,
  framexrightmargin=3mm,
  framesep=2mm,
  rulecolor=\color{subheadingcolor}
}

% Tables with text wrapping and proper column widths
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
\usepackage{tabularx}
\usepackage{calc}

% Increase row height for readability
\renewcommand{\arraystretch}{1.4}

% Allow text wrapping in table cells by default
\newcolumntype{L}[1]{>{\raggedright\arraybackslash}p{#1}}
\newcolumntype{C}[1]{>{\centering\arraybackslash}p{#1}}
\newcolumntype{R}[1]{>{\raggedleft\arraybackslash}p{#1}}

% Force Pandoc longtables to use full width and wrap text
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\footnotesize}
\makeatletter
\let\oldlt\longtable
\let\endoldlt\endlongtable
\def\longtable{\@ifnextchar[\longtable@i \longtable@ii}
\def\longtable@i[#1]{\begin{oldlt}[#1]}
\def\longtable@ii{\begin{oldlt}[c]}
\def\endlongtable{\end{oldlt}}
\makeatother

% Prevent table columns from overflowing page margins
\setlength{\tabcolsep}{4pt}

% Links
\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  linkcolor=headingcolor,
  urlcolor=linkblue,
  pdfborder={0 0 0}
}

% Page header/footer
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0.5pt}
\fancyhead[L]{\leftmark}
\fancyhead[R]{\thepage}

\usepackage{setspace}
